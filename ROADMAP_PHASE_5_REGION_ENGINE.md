```typescript
// 1. إنشاء مكون الخريطة
// client/src/components/RegionMap.tsx
import { Map } from '@/components/Map';

export function RegionMap({ regions }: { regions: Region[] }) {
  return (
    <Map
      onMapReady={(map) => {
        // رسم المناطق على الخريطة
        regions.forEach(region => {
          const color = getRegionColor(region.sentiment);
          drawRegion(map, region, color);
        });
      }}
    />
  );
}

// 2. إنشاء إحصائيات المناطق
// client/src/components/RegionStats.tsx
export function RegionStats({ regions }: { regions: Region[] }) {
  return (
    <div className="space-y-2">
      {regions.map(region => (
        <div key={region.id} className="flex items-center justify-between p-3 border rounded">
          <div>
            <p className="font-semibold">{region.name}</p>
            <p className="text-sm text-gray-500">{region.country}</p>
          </div>
          <div className="text-right">
            <p className="text-lg font-bold">{(region.sentiment * 100).toFixed(0)}%</p>
            <p className="text-xs text-gray-500">{region.eventCount} events</p>
          </div>
        </div>
      ))}
    </div>
  );
}
```
